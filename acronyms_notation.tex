\newnotation{G}{$G$}{Lie group}
\newnotation{SE}{$\text{SE}(n)$}{Special Euclidean group of dimension $n$}
\newnotation{SO}{$\text{SO}(n)$}{Special Orthogonal group of dimension $n$}
\newnotation{g}{$\mathfrak{g}$}{Lie algebra of a Lie group $G$}
\newnotation{se}{$\mathfrak{se}(n)$}{Lie algebra of the Special Euclidean group $\text{SE}(n)$}
\newnotation{so}{$\mathfrak{so}(n)$}{Lie algebra of the Special Orthogonal group $\text{SO}(n)$}
\newnotation{twist}{$\boldsymbol{\xi}$}{Gerenalized twist}
\newnotation{Smap}{$\mathcal{S}$}{Isomorphism that maps $\mathbb{R}^m\mapsto\mathfrak{g}$}
\newnotation{invSmap}{$\mathcal{S}^{-1}$}{Inverse isomorphism that maps $\mathfrak{g}\mapsto\mathbb{R}^m$, such that $\mathcal{S}^{-1}\bigl(\mathcal{S}(\boldsymbol{\xi})\bigr)=\boldsymbol{\xi}$}
\newnotation{eedist}{$\widehat{D}(\mathbf{V}, \mathbf{W})$}{Element-Element (EE-) distance between two Lie group elements $\mathbf{V}$ and $\mathbf{W}$}
\newnotation{ecdistance}{$D(\mathbf{V})$}{Element-Curve (EC-) distance between a Lie group element $\mathbf{V}$ and a curve $\mathcal{C}$}
\newnotation{Rplus}{$\mathbb{R}_+$}{Set of non-negative real numbers}
\newnotation{Rn}{$\mathbb{R}^n$}{Euclidean space of dimension $n$. The elements are $n$-dimensional column-vectors}
\newnotation{Log}{$\Log$}{Principal matrix logarithm}
\newnotation{exp}{$\exp$}{Matrix exponential}
\newnotation{tr}{$\tr$}{Matrix trace}
\newnotation{norm}{$\|\cdot\|$}{Euclidean norm}
\newnotation{frobnorm}{$\|\cdot\|_F$}{Frobenius norm}
